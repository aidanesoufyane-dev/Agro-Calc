  static Future<Uint8List> generateReport(ReportData data) async {
    final pdf = pw.Document(
      title: 'Rapport Fertigation NPK',
      author: 'AgroCalc Pro',
      subject: 'Analyse Complète de Solution Nutritive',
      creator: 'AgroCalc Pro v1.0',
    );

    pdf.addPage(pw.MultiPage(
      pageFormat: PdfPageFormat.a4,
      margin: pw.EdgeInsets.zero,
      header: (ctx) => ctx.pageNumber > 1
        ? _buildPageHeader(data, ctx.pageNumber, ctx.pagesCount)
        : pw.SizedBox(),
      footer: (ctx) => _buildPageFooter(data, ctx.pageNumber,
        ctx.pagesCount),
      build: (ctx) => [
        _buildCoverHeader(data),
        _buildBodyPadding(pw.Column(children: [
          pw.SizedBox(height: 24),
          _buildInputsSection(data),
          pw.SizedBox(height: 20),
          _buildIonicBalanceSection(data),
          pw.SizedBox(height: 20),
          _buildECSection(data),
          pw.SizedBox(height: 20),
          _buildFertUnitsSection(data),
          pw.SizedBox(height: 20),
          _buildNitrogenFormSection(data),
          if (data.distribution != null) ...[
            pw.SizedBox(height: 20),
            _buildBacDistributionSection(data.distribution!),
          ],
          pw.SizedBox(height: 20),
          _buildAgronomiqueAdviceSection(data),
        ])),
      ],
    ));

    return pdf.save();
  }

  static pw.Widget _table({
    required List<String> headers,
    required List<List<String>> rows,
    required List<int> flexes,
    PdfColor? headerBg,
    bool boldFirstCol = true,
    Set<int> accentRows = const {},
    PdfColor accentRowBg = const PdfColor(0.85, 0.95, 0.85),
  }) {
    final hBg = headerBg ?? _green900;

    return pw.Table(
      columnWidths: {
        for (int i = 0; i < flexes.length; i++)
          i: pw.FlexColumnWidth(flexes[i].toDouble()),
      },
      children: [
        pw.TableRow(
          decoration: pw.BoxDecoration(color: hBg),
          children: headers.map((h) =>
            pw.Container(
              padding: pw.EdgeInsets.symmetric(
                horizontal: 8, vertical: 7),
              child: pw.Text(h,
                textAlign: pw.TextAlign.center,
                style: _bodyBold(color: _white, size: 8.5)),
            )
          ).toList(),
        ),
        ...rows.asMap().entries.map((e) {
          final idx = e.key;
          final row = e.value;
          final isEven = idx.isEven;
          final isAccent = accentRows.contains(idx);
          PdfColor bg;
          if (isAccent) {
            bg = accentRowBg;
          } else {
            bg = isEven ? _gray100 : _white;
          }
          return pw.TableRow(
            decoration: pw.BoxDecoration(
              color: bg,
              border: pw.Border(
                bottom: pw.BorderSide(
                  color: _green100, width: 0.3)),
            ),
            children: row.asMap().entries.map((cell) {
              final isFirst = cell.key == 0;
              return pw.Container(
                padding: pw.EdgeInsets.symmetric(
                  horizontal: 8, vertical: 6.5),
                child: pw.Text(cell.value,
                  textAlign: isFirst
                    ? pw.TextAlign.left
                    : pw.TextAlign.center,
                  style: isFirst && boldFirstCol
                    ? _bodyBold(color: _dark, size: 8.5)
                    : _body(color: _dark, size: 8.5)),
              );
            }).toList(),
          );
        }),
      ],
    );
  }

    final totalCations = data.nh4 + data.k + data.ca
                       + data.h + data.mg2;
    final diff = (totalCations - totalAnions).abs();

    final rows = [
      ['NO3-  (Nitrate)',      f(data.no3),  'meq/L', 'Anion',
       'Azote nitrique'],
      ['H2PO4-  (Phosphate)',  f(data.h2po4),'meq/L', 'Anion',
       'Phosphore assimilable'],

       'Soufre'],
      ['Cl-  (Chlorure)',      f(data.clk),  'meq/L', 'Anion',
       'Indicateur salinite'],
      ['NH4+  (Ammonium)',     f(data.nh4),  'meq/L', 'Cation',
       'Azote ammoniacal'],
      ['K+  (Potassium)',      f(data.k),    'meq/L', 'Cation',
       'Fertilite / qualite'],
      ['Ca2+  (Calcium)',      f(data.ca),   'meq/L', 'Cation',
       'Structure cellulaire'],
      ['H+  (Proton)',         f(data.h),    'meq/L', 'Cation',
       'Acidite solution'],
      ['Mg2+  (Magnesium)',    f(data.mg2),  'meq/L', 'Cation',
       'Chlorophylle'],
    ];

    return pw.Column(
      crossAxisAlignment: pw.CrossAxisAlignment.start,
      children: [
        _sectionTitle(
          title: '2.  Equilibre Ionique',
          subtitle:
            'Concentration de chaque ion en meq/L dans la solution finale',
          accentColor: _green600,

        pw.Row(
          crossAxisAlignment: pw.CrossAxisAlignment.start,
          children: [
            pw.Expanded(
              flex: 60,
              child: _table(
                headers: [
                  'Ion', 'meq/L', 'Unite', 'Type', 'Role agronomique'

                rows: rows,
                flexes: [4, 2, 2, 2, 4],
                headerBg: _green900,

  static pw.Widget _ecBar(double ec) {
    final labels = ['<0.5\nT.Faible','0.5-1.5\nFaible',

      _amber700, _red900];
    int active = 0;
    if (ec >= 3.5)      active = 4;
    else if (ec >= 2.5) active = 3;
    else if (ec >= 1.5) active = 2;
    else if (ec >= 0.5) active = 1;

      children: List.generate(5, (i) =>
        pw.Expanded(
          child: pw.Container(
            height: i == active ? 34 : 26,
            margin: pw.EdgeInsets.symmetric(horizontal: 1.5),
            decoration: pw.BoxDecoration(
              color: i == active
                ? colors[i]
                : PdfColor(colors[i].red, colors[i].green, colors[i].blue, 0.2),
              borderRadius: pw.BorderRadius.circular(3),
              border: i == active
                ? pw.Border.all(color: _dark, width: 1.2)
                : pw.Border.all(color: _gray300, width: 0.5),

            child: pw.Center(
              child: pw.Text(labels[i],

  static String _ecAdviceLevel(double ec) {

  static String _ionicAdvice(ReportData data) {
    final diff =
      ((data.nh4+data.k+data.ca+data.h+data.mg2) -
       (data.no3+data.h2po4+data.so42+data.clk)).abs();
    if (diff < 0.5) return
      'Equilibre ionique excellent.\n'
      'Solution bien tamponnee.';
    if (diff < 2.0) return
      'Leger desequilibre ionique.\n'
      'Acceptable pour la plupart\ndes cultures.';
    return
      'Desequilibre ionique notable.\n'
      'Verifier les formules et\nreajuster les doses.';
  }

  static String _ionicLevel(ReportData data) {
    final diff =
      ((data.nh4+data.k+data.ca+data.h+data.mg2) -
       (data.no3+data.h2po4+data.so42+data.clk)).abs();
    if (diff < 0.5) return 'OK';
    if (diff < 2.0) return 'Surveiller';
    return 'Attention';
  }

  static Future<String> savePdfToTemp(
      Uint8List bytes, DateTime date) async {
    final dir = await getTemporaryDirectory();
    final fileName =
      'AgroCalc_Rapport_'
      '${DateFormat('yyyyMMdd_HHmm').format(date)}.pdf';
    final file = File('${dir.path}/$fileName');
    await file.writeAsBytes(bytes);
    return file.path;
  }

  static Future<void> sharePdf(
      Uint8List bytes, DateTime date) async {
    final path = await savePdfToTemp(bytes, date);
    await Share.shareXFiles(
      [XFile(path, mimeType: 'application/pdf')],
      subject: 'Rapport Fertigation NPK - AgroCalc Pro',
      text:
        'Rapport d\'analyse de solution nutritive '
        'genere par AgroCalc Pro.',
    );
  }
